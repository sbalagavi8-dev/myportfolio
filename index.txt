  <body>
    <div class="container py-5">
      <div class="text-center mb-5">
        <span class="badge bg-primary mb-3">Portfolio Visualizer</span>
        <h1 class="display-6 fw-bold">My Portfolio Holdings</h1>
        <p class="text-secondary mx-auto" style="max-width: 640px;">
          A reactive dashboard that loads your holdings and displays each stock as a modern card.
        </p>
      </div>

      <div class="row gy-3 mb-4">
        <div class="col-12 col-sm-6 col-md-3">
          <h5 class="mb-1">Total Holdings</h5>
          <p id="total-holdings" class="fs-3 fw-semibold">—</p>
        </div>
        <div class="col-12 col-sm-6 col-md-3">
          <h5 class="mb-1">Total Investment</h5>
          <p id="total-investment" class="fs-3 fw-semibold">—</p>
        </div>
        <div class="col-12 col-sm-6 col-md-3">
          <h5 class="mb-1">Current Value</h5>
          <p id="current-value" class="fs-3 fw-semibold">—</p>
        </div>
        <div class="col-12 col-sm-6 col-md-3">
          <h5 class="mb-1">P/L</h5>
          <p id="total-pl" class="fs-3 fw-semibold">—</p>
        </div>
      </div>

      

      <div class="d-flex justify-content-between align-items-center mb-4">
        <div class="d-flex align-items-center gap-2">
          <span class="text-muted small me-2">Filter:</span>
          <button id="filter-all" class="btn btn-outline-primary btn-sm">All</button>
          <button id="filter-etf" class="btn btn-outline-primary btn-sm">ETF</button>
          <button id="filter-stk" class="btn btn-outline-primary btn-sm">STK</button>
        </div>
        <div class="d-flex align-items-center gap-2">
          <span class="text-muted small me-2">Sort by:</span>
          <button id="sort-name" class="btn btn-outline-primary btn-sm">Name</button>
          <button id="sort-invested" class="btn btn-outline-primary btn-sm">Invested</button>
          <button id="sort-pl" class="btn btn-outline-primary btn-sm">P/L%</button>
          <button id="sort-toggle" class="btn btn-outline-secondary btn-sm">Asc</button>
        </div>
      </div>

      <div id="status" class="alert alert-info d-none" role="alert"></div>
      <div id="holdings-row" class="row gy-4"></div>
    </div>

    <script>
      const statusElement = document.getElementById("status");
      const holdingsRow = document.getElementById("holdings-row");
      const totalHoldings = document.getElementById("total-holdings");
      const totalInvestment = document.getElementById("total-investment");
      const currentValue = document.getElementById("current-value");
      const totalPl = document.getElementById("total-pl");
    //   const dataSource = document.getElementById("data-source");

      let currentHoldings = [];

      function showStatus(message, type = "info") {
        statusElement.textContent = message;
        statusElement.className = `alert alert-${type}`;
        statusElement.classList.remove("d-none");
      }

      function hideStatus() {
        statusElement.classList.add("d-none");
      }

      function renderCard(item, index) {
        const symbol = item.tradingSymbol || item.symbol || "Unknown";
        const rawName = item.stockName || item.companyName || item.symbolName || symbol;
        const name = rawName !== symbol ? rawName : null;
        const qty = item.totalQty ?? item.qty ?? 0;
        const avgPrice = item.avgCostPrice ?? item.averagePrice ?? 0;
        const lastPrice = item.lastPrice ?? item.closePrice ?? item.price ?? null;
        const lastPriceValue = Number(lastPrice);
        const lastPriceText = Number.isFinite(lastPriceValue) ? `₹${lastPriceValue.toFixed(2)}` : "—";
        const rawExchange = (item.exchange || item.stockExchange || "ALL").toUpperCase();
        const marketTag = rawExchange === "ALL" ? "NSE" : rawExchange.includes("BSE") ? "BSE" : "NSE";
        const instrumentType = item.instrumentType || "UNKNOWN";
        const typeTag = instrumentType === "ETF" ? "ETF" : "STK";
        const totalInvested = qty * avgPrice;
        const profitValue = Number.isFinite(lastPriceValue) ? (lastPriceValue - avgPrice) * qty : null;
        const profitPercent = Number.isFinite(lastPriceValue) && avgPrice > 0 ? ((lastPriceValue - avgPrice) / avgPrice) * 100 : null;
        const profitText = profitValue === null ? "—" : `₹${profitValue.toFixed(2)}`;
        const percentText = profitPercent === null ? "—" : `${profitPercent.toFixed(2)}%`;
        const profitClass = profitValue === null ? "text-muted" : profitValue >= 0 ? "text-success" : "text-danger";
        const themeClass = [`card-theme-1`, `card-theme-2`, `card-theme-3`, `card-theme-4`][index % 4];
        let cardClass = themeClass;

        if (profitPercent !== null) {
          if (profitPercent > 0) {
            cardClass += " profit";
          } else {
            const lossPercent = Math.abs(profitPercent);
            if (lossPercent < 10) cardClass += " loss-light";
            else if (lossPercent < 25) cardClass += " loss-medium";
            else cardClass += " loss-heavy";
          }
        }

        const progressWidth = Number.isFinite(profitPercent) ? Math.min(Math.max(profitPercent + 50, 10), 100) : 50;

        return `
          <div class="col-12 col-sm-6 col-md-6 col-xl-4">
            <div class="card h-100 shadow-sm hover-card ${cardClass}">
              <div class="card-body p-3 d-flex flex-column justify-content-between">
                <div>
                  <div class="d-flex justify-content-between align-items-start mb-2 gap-2">
                    <div>
                      <h6 class="card-title fw-bold mb-1">${symbol}</h6>
                      ${name ? `<p class="text-muted small mb-1">${name}</p>` : ""}
                    </div>
                    <div class="d-flex flex-wrap align-items-center gap-2">
                      <span class="badge tag-badge qty py-1 px-2">Qty ${qty}</span>
                      <span class="badge tag-badge inv py-1 px-2">Inv ₹${totalInvested.toFixed(0)}</span>
                      <span class="badge tag-badge market py-1 px-2">${marketTag}</span>
                      <span class="badge tag-badge type py-1 px-2">${typeTag}</span>
                    </div>
                  </div>

                  <div class="d-flex justify-content-between align-items-end mb-2 small text-muted">
                    <div>
                      <p class="mb-1">Avg Cost</p>
                      <p class="fw-semibold mb-0">₹${avgPrice.toFixed(2)}</p>
                    </div>
                    <div class="text-center">
                      <p class="mb-1">Last</p>
                      <p class="fw-semibold mb-0">${lastPriceText}</p>
                    </div>
                    <div class="text-end ${profitClass}">
                      <p class="mb-1">P/L</p>
                      <p class="fw-semibold mb-0">${profitText}</p>
                      <p class="small mb-0">${percentText}</p>
                    </div>
                  </div>
                </div>

                <div class="mt-3">
                  <div class="progress rounded-pill" style="height: 8px;">
                    <div class="progress-bar bg-gradient" role="progressbar" style="width: ${progressWidth}%"></div>
                  </div>
                </div>
              </div>
            </div>
          </div>
        `;
      }

      async function loadHoldings() {
        try {
          showStatus("Loading portfolio data…", "info");
          const response = await fetch("/api/holdings");
          const result = await response.json();

          if (!response.ok || !result.holdings) {
            showStatus("Unable to load holdings. Check your token or network.", "danger");
            currentHoldings = [];
            holdingsRow.innerHTML = "";
            updateTotals();
            dataSource.textContent = "error";
            return;
          }

          const holdings = result.holdings;
        //   dataSource.textContent = result.source || "unknown";
          currentHoldings = holdings;

          if (holdings.length === 0) {
            showStatus("No holdings were returned. Add holdings to your portfolio.", "warning");
            updateTotals();
            return;
          }

          hideStatus();
          updateTotals();
          filterAndSortHoldings();
        } catch (error) {
          showStatus("Failed to fetch holdings. Open browser console for details.", "danger");
          console.error(error);
        }
      }

      let sortField = 'name'; // default sort field
      let sortAsc = true; // true for ascending, false for descending
      let currentFilter = 'all'; // 'all', 'etf', 'stk'

      function updateTotals() {
        let filtered = [...currentHoldings];
        
        // Filter
        if (currentFilter === 'etf') {
          filtered = filtered.filter(item => item.instrumentType === 'ETF');
        } else if (currentFilter === 'stk') {
          filtered = filtered.filter(item => item.instrumentType === 'STK');
        }

        const count = filtered.length;
        const investment = filtered.reduce((sum, item) => sum + (item.totalInvested || 0), 0);
        const value = filtered.reduce((sum, item) => {
          const qty = item.totalQty || 0;
          const price = item.lastPrice || 0;
          return sum + (qty * price);
        }, 0);
        const pl = filtered.reduce((sum, item) => sum + (item.profit || 0), 0);

        totalHoldings.textContent = count;
        totalInvestment.textContent = `₹${investment.toFixed(0)}`;
        currentValue.textContent = `₹${value.toFixed(0)}`;
        totalPl.textContent = pl >= 0 ? `₹${pl.toFixed(0)}` : `-₹${Math.abs(pl).toFixed(0)}`;
        totalPl.className = pl >= 0 ? 'fs-3 fw-semibold text-success' : 'fs-3 fw-semibold text-danger';
      }

      function updateButtons() {
        // Update sort buttons
        const sortButtons = {
          name: document.getElementById('sort-name'),
          invested: document.getElementById('sort-invested'),
          pl: document.getElementById('sort-pl'),
        };

        Object.entries(sortButtons).forEach(([field, button]) => {
          if (!button) return;
          if (sortField === field) {
            button.classList.remove('btn-outline-primary');
            button.classList.add('btn-primary');
          } else {
            button.classList.remove('btn-primary');
            button.classList.add('btn-outline-primary');
          }
        });

        const toggleButton = document.getElementById('sort-toggle');
        toggleButton.textContent = sortAsc ? 'Asc' : 'Desc';
        toggleButton.classList.toggle('btn-secondary', !sortAsc);
        toggleButton.classList.toggle('btn-outline-secondary', sortAsc);

        // Update filter buttons
        const filterButtons = {
          all: document.getElementById('filter-all'),
          etf: document.getElementById('filter-etf'),
          stk: document.getElementById('filter-stk'),
        };

        Object.entries(filterButtons).forEach(([filter, button]) => {
          if (!button) return;
          if (currentFilter === filter) {
            button.classList.remove('btn-outline-primary');
            button.classList.add('btn-primary');
          } else {
            button.classList.remove('btn-primary');
            button.classList.add('btn-outline-primary');
          }
        });
      }

      function filterAndSortHoldings() {
        updateButtons();
        updateTotals();
        let filtered = [...currentHoldings];
        
        // Filter
        if (currentFilter === 'etf') {
          filtered = filtered.filter(item => item.instrumentType === 'ETF');
        } else if (currentFilter === 'stk') {
          filtered = filtered.filter(item => item.instrumentType === 'STK');
        }
        
        // Sort
        filtered.sort((a, b) => {
          let aVal, bVal;
          switch (sortField) {
            case 'name':
              aVal = (a.stockName || a.tradingSymbol || "").toLowerCase();
              bVal = (b.stockName || b.tradingSymbol || "").toLowerCase();
              break;
            case 'invested':
              aVal = parseFloat(a.totalInvested || 0);
              bVal = parseFloat(b.totalInvested || 0);
              break;
            case 'pl':
              aVal = parseFloat(a.profitPercent || 0);
              bVal = parseFloat(b.profitPercent || 0);
              break;
            default:
              return 0;
          }
          if (sortAsc) {
            return aVal > bVal ? 1 : aVal < bVal ? -1 : 0;
          } else {
            return aVal < bVal ? 1 : aVal > bVal ? -1 : 0;
          }
        });
        holdingsRow.innerHTML = filtered.map((item, index) => renderCard(item, index)).join("");
      }

      // Event listeners for sort buttons
      document.getElementById('sort-name').addEventListener('click', () => {
        sortField = 'name';
        filterAndSortHoldings();
      });
      document.getElementById('sort-invested').addEventListener('click', () => {
        sortField = 'invested';
        filterAndSortHoldings();
      });
      document.getElementById('sort-pl').addEventListener('click', () => {
        sortField = 'pl';
        filterAndSortHoldings();
      });
      document.getElementById('sort-toggle').addEventListener('click', () => {
        sortAsc = !sortAsc;
        filterAndSortHoldings();
      });

      // Event listeners for filter buttons
      document.getElementById('filter-all').addEventListener('click', () => {
        currentFilter = 'all';
        filterAndSortHoldings();
      });
      document.getElementById('filter-etf').addEventListener('click', () => {
        currentFilter = 'etf';
        filterAndSortHoldings();
      });
      document.getElementById('filter-stk').addEventListener('click', () => {
        currentFilter = 'stk';
        filterAndSortHoldings();
      });

      loadHoldings();
    </script>
